# Hay tres tipos de numeros en Python: "int", "float" y "complex".
x = 1    # Es int
y = 2.8  # Es float
z = 1j   # Es complex

print (type(x))
print (type(y))
print (type(z))
print (" ")

# "int" es un numero entero positivo o negativo.
a = 1
b = 35656222554887711
c = -3255522

print(type(a))
print(type(b))
print(type(c))
print (" ")

# "float" es un numero decimal positivo o negativo.
w = 1.10
a = 1.0
s = -35.59

print(type(w))
print(type(a))
print(type(s))
print (" ")

# "complex" es un numero positivo o negativo que tiene una "e" consigo, lo cual indica una potencia de diez.
p = 35e3
l = 12E4
m = -87.7e100

print(type(p))
print(type(l))
print(type(m))
print (" ")
